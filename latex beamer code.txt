% Slide 3: Forces
\begin{frame}{The Physics Model: Force Analysis}
    The dynamics of the jump are governed by three primary forces:
    
    \vspace{0.4cm}
    \begin{itemize}
        \item \textbf{Gravity (weight):} Pulls the jumper downward.
        \begin{equation} F_g=mg \end{equation}
        \item \textbf{Elastic force (tension):} Cord pulls upward when stretched.
        \begin{equation} F_R=kx \end{equation}
        \item \textbf{Air resistance (drag):} Opposes motion.
        \begin{equation} F_d=-bv \end{equation}
        \textbf{Net Force Equation:}
            \[ \sum F = F_g + F_d + F_s \]
    \end{itemize}
\end{frame}

% Slide 4: The Physics Model - 3 Stages of the Jump
\begin{frame}{The Physics Model: 3 Stages of the Jump}
    The motion is divided into phases based on the state of the bungee cord:
    
    \vspace{0.3cm}
    \begin{enumerate}
        \item \textbf{Phase 1: Free Fall ($y < L$)} \\
        The cord is slack; Hooke's Law is inactive ($F_s = 0$).
        
        \vspace{0.4cm}
        \item \textbf{Phase 2: Cord Stretching ($y \geq L$)} \\
        Hooke's Law activates; upward restoring force $F_s$ begins to decelerate the jumper.
        
        \vspace{0.4cm}
        \item \textbf{Phase 3: Maximum Extension ($v = 0$)} \\
        Peak stretch ($x_{max}$) and peak upward acceleration (Max G-Force).
    \end{enumerate}
\end{frame}

% Slide 5: The Differential Equations of Motion
\begin{frame}{The Differential Equations of Motion}
    To simulate the 1D motion, we break the second-order system into two coupled first-order differential equations:
    
    \vspace{0.6cm}
    
    % Centered Equations
    \centering
    {\large \textbf{1. Rate of Change of Position}}
    \begin{equation*}
        \frac{dy}{dt} = v
    \end{equation*}

    \vspace{0.4cm}

    {\large \textbf{2. Rate of Change of Velocity}}
    \begin{equation*}
        \frac{dv}{dt} = a = \frac{F_{net}}{m}
    \end{equation*}

    \vspace{0.8cm}
    \flushleft
    \textbf{Dynamics of $F_{net}$:}
    \begin{itemize}
        \item \textbf{Initial Fall ($y \leq L$):} $F_{net} = mg - bv$
        \item \textbf{Cord Stretching ($y > L$):} $F_{net} = mg - bv - k(y-L)$
    \end{itemize}
    
    \vspace{0.2cm}
    \centering
  
\end{frame}
% Slide 5: Numerical Integration (The Euler Loop)
\begin{frame}[fragile]{Numerical Integration: The Euler Loop}
    The simulation updates the state variables in discrete time steps ($dt$) using the Euler Method:
\vspace{0.3cm}
    \begin{center}
    \begin{minipage}{0.9\textwidth}
    \begin{lstlisting}[language=Python, basicstyle=\ttfamily\small, keywordstyle=\color{brown!80!black}]
while jumper.pos.y > stop_height:
    rate(100)
    
    # 1. Calculate Net Force (Conditional)
    Fnet = m * g_vec - b * jumper.vel
    if jumper.pos.y < (start_height - L):
        Fnet += -k * stretch_vec
        
    # 2. Update Velocity: v = v + a*dt
    jumper.vel += (Fnet / m) * dt
    
    # 3. Update Position: y = y + v*dt
    jumper.pos += jumper.vel * dt
    \end{lstlisting}
    \end{minipage}
    \end{center}

    \vspace{0.2cm}
    \centering
    \textit{\small The loop continues until the jumper reaches the lowest point or stabilizes.}
\end{frame}

% Slide 5: Calculation (Stretch)
\begin{frame}{Calculating Maximum Cord Stretch}
    By applying the Law of Conservation of Energy:
    
    \begin{block}{Energy Conversion Formula}
    At the lowest point, gravitational potential energy becomes elastic potential energy:
    \begin{equation} mg(L+x) = \frac{1}{2}kx_{max}^2 \end{equation}
    \vspace{0.2cm}
    Solving for $x_{max}$:
    \begin{equation} x_{max} = \sqrt{\frac{2mg(L+x)}{k}} \end{equation}
    \end{block}
\end{frame}

% Slide 6: Calculation (G-Force)
\begin{frame}{Calculating Maximum G-Force}
    Using Newton's Second Law at the peak of the stretch:
    
    \vspace{0.2cm}
    \centering $F_{net} = ma \implies kx_{max} - mg = ma$
    
    \begin{block}{Derivation of G-Force}
    \begin{equation} a = \frac{kx_{max} - mg}{m} \end{equation}
    \begin{equation} \text{g-force} = \frac{a}{g} = \frac{kx_{max}}{mg} \end{equation}
    \end{block}
\end{frame}
